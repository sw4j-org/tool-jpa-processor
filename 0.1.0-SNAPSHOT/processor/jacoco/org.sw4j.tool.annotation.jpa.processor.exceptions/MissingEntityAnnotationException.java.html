package org.sw4j.tool.annotation.jpa.processor.exceptions;

import javax.annotation.Nonnull;

/**
 * This is an exception that indicates that the processed class is not annotated with the {@code @Entity} annotation.
 *
 * @author Uwe Plonus
 */
public class MissingEntityAnnotationException extends AnnotationProcessorException {

    /**
     * A constructor with a class name to refine this exception.
     *
     * @param className the name of the class that causes this exception.
     */
    public MissingEntityAnnotationException(@Nonnull final String className) {
        super(new StringBuilder("The processed class \"").append(className)
                .append("\" is not an entity.").toString());
    }

    /**
     * A constructor with a class name to refine this exception and a cause of this exception.
     *
     * @param className the name of the class that causes this exception.
     * @param cause the throwable that causes this exception.
     */
    public MissingEntityAnnotationException(@Nonnull final String className, @Nonnull final Throwable cause) {
        super(new StringBuilder("The processed class \"").append(className)
                .append("\" is not an entity.").toString(), cause);
    }

}
